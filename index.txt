    <script>
        // Dados Iniciais salvos no LocalStorage (Se vazio, inicia com a nota do print)
        let notas = JSON.parse(localStorage.getItem('notas_fiscais')) || [
            {
                id: "1",
                data: "2026-06-29",
                fornecedor: "Apol",
                nf: "61.975",
                valor: 1119.13,
                chegada: "07:06",
                saida: "07:10",
                temp: "-",
                conferente: "Amanda",
                prevencao: "Lucas",
                status: "ACATADA",
                observacao: "-"
            }
        ];

        // Formatar valores para Moeda Brasileira
        function formatarMoeda(valor) {
            return new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' }).format(valor);
        }

        // Formatar datas do input (AAAA-MM-DD) para padrão BR (DD/MM/AAAA)
        function formatarDataBR(dataIso) {
            if(!dataIso) return '-';
            const [ano, mes, dia] = dataIso.split('-');
            return `${dia}/${mes}/${ano}`;
        }

        // Renderiza e atualiza a interface baseada nos dados atuais
        function renderizarTabela() {
            const corpo = document.getElementById('tabela-corpo');
            corpo.innerHTML = '';

            // Filtros ativos
            const fFornecedor = document.getElementById('filtro-fornecedor').value.toLowerCase();
            const fNf = document.getElementById('filtro-nf').value.toLowerCase();
            const fStatus = document.getElementById('filtro-status').value;

            let notasFiltradas = notas.filter(nota => {
                return (nota.fornecedor.toLowerCase().includes(fFornecedor)) &&
                       (nota.nf.toLowerCase().includes(fNf)) &&
                       (fStatus === "" || nota.status === fStatus);
            });

            // Atualiza Contadores do Painel Superior
            document.getElementById('card-total-notas').innerText = notasFiltradas.length;
            
            const totalValor = notasFiltradas.reduce((acc, nota) => acc + parseFloat(nota.valor || 0), 0);
            document.getElementById('card-valor-total').innerText = formatarMoeda(totalValor);

            const qtdAcatadas = notasFiltradas.filter(n => n.status === 'ACATADA').length;
            document.getElementById('card-acatadas').innerText = qtdAcatadas;

            const qtdNaoAcatadas = notasFiltradas.filter(n => n.status === 'NÃO ACATADA').length;
            document.getElementById('card-nao-acatadas').innerText = qtdNaoAcatadas;

            document.getElementById('contador-registros').innerText = `${notasFiltradas.length} ${notasFiltradas.length === 1 ? 'nota encontrada' : 'notas encontradas'}`;

            // Criar as linhas da Tabela
            notasFiltradas.forEach(nota => {
                const tr = document.createElement('tr');
                tr.className = "hover:bg-gray-50/80 transition text-gray-600 border-b border-gray-100";
                
                const statusClasse = nota.status === 'ACATADA' 
                    ? 'bg-green-50 text-green-700 border border-green-200' 
                    : 'bg-red-50 text-red-700 border border-red-200';

                tr.innerHTML = `
                    <td class="p-4 text-gray-800">${formatarDataBR(nota.data)}</td>
                    <td class="p-4 font-semibold text-gray-800">${nota.fornecedor}</td>
                    <td class="p-4">${nota.nf}</td>
                    <td class="p-4 font-bold text-gray-800">${formatarMoeda(nota.valor)}</td>
                    <td class="p-4">${nota.chegada || '-'}</td>
                    <td class="p-4">${nota.saida || '-'}</td>
                    <td class="p-4">${nota.temp || '-'}</td>
                    <td class="p-4">${nota.conferente || '-'}</td>
                    <td class="p-4">${nota.prevencao || '-'}</td>
                    <td class="p-4">
                        <span class="px-2.5 py-1 text-[11px] font-bold rounded-md tracking-wide uppercase ${statusClasse}">
                            ${nota.status}
                        </span>
                    </td>
                    <td class="p-4 text-xs text-gray-400">${nota.observacao || '-'}</td>
                    <td class="p-4 text-center">
                        <div class="flex justify-center gap-3">
                            <button onclick="editarNota('${nota.id}')" class="text-amber-500 hover:text-amber-600 transition" title="Editar">
                                <i class="fa-solid fa-pencil text-base"></i>
                            </button>
                            <button onclick="excluirNota('${nota.id}')" class="text-gray-400 hover:text-red-500 transition" title="Excluir">
                                <i class="fa-solid fa-trash-can text-base"></i>
                            </button>
                        </div>
                    </td>
                `;
                corpo.appendChild(tr);
            });

            // Sincroniza as alterações no armazenamento local do navegador
            localStorage.setItem('notas_fiscais', JSON.stringify(notas));
        }

        // Funções do Modal
        function abrirModal() {
            document.getElementById('modal-titulo').innerText = "Nova Nota Fiscal";
            document.getElementById('form-nota').reset();
            document.getElementById('nota-id').value = "";
            document.getElementById('modal-nota').classList.remove('hidden');
        }

        function fecharModal() {
            document.getElementById('modal-nota').classList.add('hidden');
        }

        // Salvar Nova Nota ou Salvar Edição
        function salvarNota(event) {
            event.preventDefault();
            const id = document.getElementById('nota-id').value;

            const dadosNota = {
                id: id || Date.now().toString(),
                data: document.getElementById('input-data').value,
                fornecedor: document.getElementById('input-fornecedor').value,
                nf: document.getElementById('input-nf').value,
                valor: parseFloat(document.getElementById('input-valor').value),
                chegada: document.getElementById('input-chegada').value,
                saida: document.getElementById('input-saida').value,
                temp: document.getElementById('input-temp').value || "-",
                conferente: document.getElementById('input-conferente').value,
                prevencao: document.getElementById('input-prevencao').value,
                status: document.getElementById('input-status').value,
                observacao: document.getElementById('input-observacao').value || "-"
            };

            if (id) {
                // Modo Edição
                const index = notas.findIndex(n => n.id === id);
                if(index !== -1) notas[index] = dadosNota;
            } else {
                // Modo Criação
                notas.push(dadosNota);
            }

            fecharModal();
            renderizarTabela();
        }

        // Mapeia os dados da linha selecionada de volta para os inputs do modal
        function editarNota(id) {
            const nota = notas.find(n => n.id === id);
            if (!nota) return;

            document.getElementById('modal-titulo').innerText = "Editar Nota Fiscal";
            document.getElementById('nota-id').value = nota.id;
            document.getElementById('input-data').value = nota.data;
            document.getElementById('input-fornecedor').value = nota.fornecedor;
            document.getElementById('input-nf').value = nota.nf;
            document.getElementById('input-valor').value = nota.valor;
            document.getElementById('input-chegada').value = nota.chegada;
            document.getElementById('input-saida').value = nota.saida;
            document.getElementById('input-temp').value = nota.temp;
            document.getElementById('input-conferente').value = nota.conferente;
            document.getElementById('input-prevencao').value = nota.prevencao;
            document.getElementById('input-status').value = nota.status;
            document.getElementById('input-observacao').value = nota.observacao;

            document.getElementById('modal-nota').classList.remove('hidden');
        }

        // Remover nota do array principal
        function excluirNota(id) {
            if(confirm("Tem certeza que deseja excluir esta nota fiscal?")) {
                notas = notas.filter(n => n.id !== id);
                renderizarTabela();
            }
        }

        function limparFiltros() {
            document.getElementById('filtro-fornecedor').value = "";
            document.getElementById('filtro-nf').value = "";
            document.getElementById('filtro-status').value = "";
            renderizarTabela();
        }

        // Lógica para exportar os dados visíveis como arquivo .csv
        function exportarCSV() {
            let csvContent = "data:text/csv;charset=utf-8,";
            csvContent += "Data,Fornecedor,NF,Valor,Chegada,Saida,Temp,Conferente,Prevencao,Status,Observacao\n";

            notas.forEach(n => {
                let linha = `"${formatarDataBR(n.data)}","${n.fornecedor}","${n.nf}","${n.valor}","${n.chegada}","${n.saida}","${n.temp}","${n.conferente}","${n.prevencao}","${n.status}","${n.observacao}"`;
                csvContent += linha + "\n";
            });

            const encodedUri = encodeURI(csvContent);
            const link = document.createElement("a");
            link.setAttribute("href", encodedUri);
            link.setAttribute("download", "notas_fiscais.csv");
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }

        // Executar ao abrir a tela pela primeira vez
        renderizarTabela();
    </script>
